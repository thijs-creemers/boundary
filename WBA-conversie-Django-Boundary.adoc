=== Architectuur: Functional Core / Imperative Shell

Boundary enforceert strikte scheiding van concerns:

[source,text]
----
┌─────────────────────────────────────────────┐
│         IMPERATIVE SHELL (shell/*)          │
│  I/O, validation, logging, side effects     │
└─────────────────────────────────────────────┘
                    ↓ calls
┌─────────────────────────────────────────────┐
│           PORTS (ports.clj)                 │
│  Protocol definitions (interfaces)          │
└─────────────────────────────────────────────┘
                    ↑ implements
┌─────────────────────────────────────────────┐
│         FUNCTIONAL CORE (core/*)            │
│  Pure functions, business logic only        │
└─────────────────────────────────────────────┘
----

**Dependency Rules:**

* ✅ Shell → Core (shell roept core aan)
* ✅ Core → Ports (core hangt af van protocols)
* ✅ Shell → Adapters (shell implementeert protocols)
* ❌ Core → Shell (NOOIT - core is puur)
* ❌ Core → Adapters (NOOIT - geen directe I/O)

[source,clojure]

;; Pure function - geen side effects
(defn calculate-workvisit-status 
  "Bepaal status op basis van taken. Pure function."
  [workvisit tasks]
  (cond
    (every? completed? tasks) :completed
    (some in-progress? tasks) :in-progress
    (some pending? tasks) :pending
    :else :draft))

;; Shell layer doet I/O
(defn update-workvisit-status! [workvisit-svc workvisit-id]
  (let [workvisit (find-workvisit workvisit-svc workvisit-id)
        tasks (find-tasks-for-workvisit task-svc workvisit-id)
        new-status (calculate-workvisit-status workvisit tasks)]
    (update-workvisit! workvisit-svc workvisit-id {:status new-status})))

==== PostgreSQL Full-Text Search Alternatief

PostgreSQL heeft uitstekende full-text search capabilities die dit volledig kunnen vervangen:

[source,sql]
----
-- Setup (eenmalig)
CREATE EXTENSION IF NOT EXISTS pg_trgm;  -- Voor fuzzy matching

-- GIN index voor snelle full-text search
CREATE INDEX workvisit_search_idx ON workvisits 
  USING GIN(
    setweight(to_tsvector('dutch', event_name), 'A') ||
    setweight(to_tsvector('dutch', event_city), 'A') ||
    setweight(to_tsvector('dutch', coalesce(motivation, '')), 'C')
  );

-- Trigram index voor partial matching
CREATE INDEX workvisit_trigram_idx ON workvisits
  USING GIN(event_name gin_trgm_ops);
----

[source,clojure]
----
;; Boundary search implementation
(defn search-workvisits 
  "Full-text search met relevance ranking."
  [db search-term {:keys [limit offset filters]}]
  (jdbc/execute! 
    db
    ["SELECT w.*, 
            o.first_name as official_first_name,
            o.last_name as official_last_name,
            s.label as status_label,
            ts_rank(
              setweight(to_tsvector('dutch', w.event_name), 'A') ||
              setweight(to_tsvector('dutch', w.event_city), 'A') ||
              setweight(to_tsvector('dutch', o.first_name || ' ' || o.last_name), 'B') ||
              setweight(to_tsvector('dutch', coalesce(w.motivation, '')), 'C') ||
              setweight(to_tsvector('dutch', coalesce(w.summary, '')), 'D'),
              plainto_tsquery('dutch', ?)
            ) as rank
      FROM workvisits w
      LEFT JOIN officials o ON w.official_id = o.id
      LEFT JOIN workvisit_statuses s ON w.status_id = s.id
      WHERE to_tsvector('dutch', 
                        w.event_name || ' ' || 
                        w.event_city || ' ' || 
                        coalesce(o.first_name, '') || ' ' || 
                        coalesce(o.last_name, '') || ' ' ||
                        coalesce(w.motivation, '')
                       ) @@ plainto_tsquery('dutch', ?)
      ORDER BY rank DESC
      LIMIT ? OFFSET ?"
     search-term search-term limit offset]))
----

  (:require [clj-http.client :as http]
            [boundary.core.utils.env :as env]
            [clojure.data.json :as json]))

(defn google-geocode 
  "Geocode adres naar lat/long via Google Maps API."
  [address]
  (let [api-key (env/get-required "GOOGLE_MAPS_API_KEY")
        response (http/get "https://maps.googleapis.com/maps/api/geocode/json"
                          {:query-params {:address address :key api-key}
                           :as :json})]
    (when-let [result (first (get-in response [:body :results]))]
      (let [location (get-in result [:geometry :location])]
        {:latitude (:lat location)
         :longitude (:lng location)
         :formatted-address (:formatted_address result)}))))

(defn update-location-coordinates 
  "Update location coordinaten als adres gewijzigd."
  [location]
  (when (:geo-auto-update location)
    (let [address (compute-address location)]
      (when (and address 
                 (not= address (:geo-last-address location)))
        (when-let [coords (google-geocode address)]
          (merge location 
                 coords
                 {:geo-last-address address}))))))

==== Schema Setup

[source,sql]
----
-- Extensions
CREATE EXTENSION IF NOT EXISTS pg_trgm;  -- Trigram matching
CREATE EXTENSION IF NOT EXISTS unaccent; -- Remove accents for search

-- Search configuration for Dutch
CREATE TEXT SEARCH CONFIGURATION nl (COPY = dutch);

-- Full-text search index (primary)
CREATE INDEX workvisit_search_idx ON workvisits 
  USING GIN(
    setweight(to_tsvector('nl', event_name), 'A') ||
    setweight(to_tsvector('nl', event_city), 'A') ||
    setweight(to_tsvector('nl', coalesce(motivation, '')), 'C') ||
    setweight(to_tsvector('nl', coalesce(summary, '')), 'D')
  );

-- Trigram index voor partial/fuzzy matching
CREATE INDEX workvisit_event_name_trgm_idx ON workvisits 
  USING GIN(event_name gin_trgm_ops);

CREATE INDEX workvisit_event_city_trgm_idx ON workvisits 
  USING GIN(event_city gin_trgm_ops);

-- Index op status/filters voor snelle filtering
CREATE INDEX workvisit_status_date_idx ON workvisits (status_id, date);
CREATE INDEX workvisit_official_idx ON workvisits (official_id);
----

==== Search Query Implementation

[source,clojure]
----
(ns boundary.workvisit.shell.search
  (:require [next.jdbc :as jdbc]
            [next.jdbc.sql :as sql]
            [boundary.core.utils.case-conversion :as cc]))

(defn search-workvisits
  "Full-text search met Dutch language support en relevance ranking.
  
  Options:
  - :limit - Max resultaten (default 100)
  - :offset - Pagination offset (default 0)
  - :filters - Map van filters {:status-id X :date-from Y :date-to Z}
  - :min-rank - Minimum relevance score (default 0.01)"
  [db search-term {:keys [limit offset filters min-rank]
                   :or {limit 100 offset 0 min-rank 0.01}}]
  (let [base-query
        "SELECT 
           w.id,
           w.slug,
           w.event_name,
           w.event_city,
           w.date,
           w.status_id,
           o.id as official_id,
           o.first_name as official_first_name,
           o.last_name as official_last_name,
           s.label as status_label,
           s.color as status_color,
           ts_rank(
             setweight(to_tsvector('nl', w.event_name), 'A') ||
             setweight(to_tsvector('nl', w.event_city), 'A') ||
             setweight(to_tsvector('nl', o.first_name || ' ' || o.last_name), 'B') ||
             setweight(to_tsvector('nl', coalesce(w.motivation, '')), 'C') ||
             setweight(to_tsvector('nl', coalesce(w.summary, '')), 'D'),
             plainto_tsquery('nl', ?)
           ) as rank
         FROM workvisits w
         LEFT JOIN officials o ON w.official_id = o.id
         LEFT JOIN workvisit_statuses s ON w.status_id = s.id
         WHERE to_tsvector('nl',
                           w.event_name || ' ' ||
                           w.event_city || ' ' ||
                           coalesce(o.first_name, '') || ' ' ||
                           coalesce(o.last_name, '') || ' ' ||
                           coalesce(w.motivation, '') || ' ' ||
                           coalesce(w.summary, '')
                          ) @@ plainto_tsquery('nl', ?)
           AND ts_rank(...) > ?"]
    
    ;; Build WHERE clause voor filters
    (let [where-clauses (cond-> []
                          (:status-id filters)
                          (conj "w.status_id = ?")
                          
                          (:date-from filters)
                          (conj "w.date >= ?")
                          
                          (:date-to filters)
                          (conj "w.date <= ?")
                          
                          (:official-id filters)
                          (conj "w.official_id = ?"))
          
          full-query (str base-query
                         (when (seq where-clauses)
                           (str " AND " (clojure.string/join " AND " where-clauses)))
                         " ORDER BY rank DESC, w.date DESC"
                         " LIMIT ? OFFSET ?")
          
          params (cond-> [search-term search-term min-rank]
                   (:status-id filters) (conj (:status-id filters))
                   (:date-from filters) (conj (:date-from filters))
                   (:date-to filters) (conj (:date-to filters))
                   (:official-id filters) (conj (:official-id filters))
                   true (conj limit offset))]
      
      (->> (jdbc/execute! db (into [full-query] params))
           (map cc/snake-case->kebab-case-map)))))

(defn search-by-slug
  "Direct slug lookup voor snelle redirect.
  Retourneert workvisit indien slug exact matcht."
  [db slug]
  (sql/get-by-id db :workvisits slug {:id-key :slug}))

==== Tenant Detection

[source,clojure]
----
(ns boundary.workvisit.shell.http
  (:require [boundary.tenant.core.tenant :as tenant-core]
            [boundary.tenant.shell.middleware :as tenant-mw]))

;; Interceptor voor tenant detection
(def tenant-interceptor
  {:name ::tenant-detection
   :enter (fn [ctx]
            (let [request (:request ctx)
                  host (get-in request [:headers "host"])
                  tenant-id (tenant-core/extract-tenant-from-host host)]
              (if (tenant-core/valid-tenant? tenant-id)
                (assoc-in ctx [:request :tenant-id] tenant-id)
                (throw (ex-info "Invalid tenant"
                               {:type :invalid-tenant
                                :host host})))))})

;; Database query met tenant isolation
(defn find-workvisit-for-tenant
  "Haal workvisit op binnen tenant schema."
  [db-context tenant-id workvisit-id]
  (tenant-mw/with-tenant-schema db-context tenant-id
    (sql/get-by-id db-context :workvisits workvisit-id)))
----

(ns boundary.workvisit.migrations
  (:require [boundary.tenant.shell.migrations :as tenant-mig]))

;; Migratie definitie (Ragtime format)
(defn up []
  ["CREATE TABLE workvisits (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      slug VARCHAR(255) NOT NULL,
      event_name VARCHAR(500) NOT NULL,
      date DATE NOT NULL,
      status_id INTEGER REFERENCES workvisit_statuses(id),
      official_id INTEGER REFERENCES officials(id),
      created_at TIMESTAMP NOT NULL DEFAULT NOW(),
      updated_at TIMESTAMP NOT NULL DEFAULT NOW()
   )"
   "CREATE INDEX workvisit_date_idx ON workvisits(date)"
   "CREATE INDEX workvisit_status_idx ON workvisits(status_id)"])

(defn down []
  ["DROP TABLE IF EXISTS workvisits CASCADE"])

;; Migratie uitvoeren voor alle tenants
(defn migrate-all-tenants! [db-context]
  (tenant-mig/migrate-all-tenants! 
    db-context
    {:migrations [{:id "001-create-workvisits"
                   :up up
                   :down down}]}))
